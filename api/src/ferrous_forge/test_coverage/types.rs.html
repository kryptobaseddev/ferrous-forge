1//! Test coverage types and configuration

3use serde::{Deserialize, Serialize};
4use std::collections::HashMap;
5
6/// Test coverage configuration
7#[derive(Debug, Clone, Serialize, Deserialize)]
8pub struct CoverageConfig {
9    /// Minimum line coverage percentage (0-100)
10    pub min_line_coverage: f64,
11    /// Minimum function coverage percentage (0-100)
12    pub min_function_coverage: f64,
13    /// Minimum branch coverage percentage (0-100)
14    pub min_branch_coverage: f64,
15    /// Whether to fail builds on coverage below threshold
16    pub fail_on_low_coverage: bool,
17    /// Files to exclude from coverage analysis
18    pub exclude_files: Vec<String>,
19    /// Directories to exclude from coverage analysis
20    pub exclude_dirs: Vec<String>,
21}
22
23impl Default for CoverageConfig {
24    fn default() -> Self {
25        Self {
26            min_line_coverage: 80.0,
27            min_function_coverage: 85.0,
28            min_branch_coverage: 75.0,
29            fail_on_low_coverage: true,
30            exclude_files: vec![
31                "main.rs".to_string(),
32                "lib.rs".to_string(),
33                "**/tests/**".to_string(),
34                "**/benches/**".to_string(),
35            ],
36            exclude_dirs: vec![
37                "target".to_string(),
38                "tests".to_string(),
39                "benches".to_string(),
40                "examples".to_string(),
41            ],
42        }
43    }
44}

46/// Test coverage results
47#[derive(Debug, Clone, Serialize, Deserialize)]
48pub struct CoverageReport {
49    /// Overall line coverage percentage
50    pub line_coverage: f64,
51    /// Overall function coverage percentage  
52    pub function_coverage: f64,
53    /// Overall branch coverage percentage
54    pub branch_coverage: f64,
55    /// Per-file coverage breakdown
56    pub file_coverage: HashMap<String, FileCoverage>,
57    /// Total lines tested
58    pub lines_tested: u32,
59    /// Total lines in codebase
60    pub total_lines: u32,
61    /// Functions tested
62    pub functions_tested: u32,
63    /// Total functions
64    pub total_functions: u32,
65    /// Branches tested
66    pub branches_tested: u32,
67    /// Total branches
68    pub total_branches: u32,
69}

71/// Coverage information for a single file

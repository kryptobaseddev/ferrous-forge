197/// Commit-msg hook script content
198pub const COMMIT_MSG_HOOK: &str = r#"#!/bin/bash
199# Ferrous Forge commit-msg hook
200# Validates commit message format
201
202commit_msg_file=$1
203commit_msg=$(cat "$commit_msg_file")
204
205# Check for conventional commit format
206if ! echo "$commit_msg" | \
207    grep -qE "^(feat|fix|docs|style|refactor|test|chore)(\(.+\))?: .{1,50}"; then
208    echo ""
209    echo "❌ Invalid commit message format"
210    echo ""
211    echo "Use conventional commit format: type(scope): description"
212    echo ""
213    echo "Examples:"
214    echo "  feat: add new validation rule"
215    echo "  fix: resolve unwrap detection bug"
216    echo "  docs: update README"
217    echo ""
218    exit 1
219fi
220
221echo "✅ Commit message format valid"
222"#;

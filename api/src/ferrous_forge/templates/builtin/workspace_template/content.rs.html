1//! Workspace template file content generation

3use super::templates::{cli_main_content, core_lib_content, utils_lib_content};
4use std::collections::HashMap;
5
6/// Create the files for workspace template
7pub fn create_workspace_files() -> HashMap<String, String> {
8    let mut files = HashMap::new();
9
10    files.insert("Cargo.toml".to_string(), workspace_cargo_toml());
11    files.insert("README.md".to_string(), readme_content());
12    files.insert("core/Cargo.toml".to_string(), core_cargo_toml());
13    files.insert("core/src/lib.rs".to_string(), core_lib_content());
14    files.insert("cli/Cargo.toml".to_string(), cli_cargo_toml());
15    files.insert("cli/src/main.rs".to_string(), cli_main_content());
16    files.insert("utils/Cargo.toml".to_string(), utils_cargo_toml());
17    files.insert("utils/src/lib.rs".to_string(), utils_lib_content());
18    files.insert(
19        ".ferrous-forge/config.toml".to_string(),
20        config_toml_content(),
21    );
22
23    files
24}
25
26/// Workspace Cargo.toml content
27fn workspace_cargo_toml() -> String {
28    r#"[workspace]
29resolver = "2"
30members = ["core", "cli", "utils"]
31
32[workspace.package]
33version = "0.1.0"
34edition = "2024"
35authors = ["{{author}}"]
36license = "MIT OR Apache-2.0"
37repository = "https://github.com/{{author}}/{{workspace_name}}"
38
39[workspace.dependencies]
40# Shared dependencies across workspace members
41anyhow = "1.0"
42thiserror = "1.0"
43serde = { version = "1.0", features = ["derive"] }
44tokio = { version = "1.40", features = ["full"] }
45clap = { version = "4.5", features = ["derive"] }
46
47# Dev dependencies
48tempfile = "3.10"
49assert_cmd = "2.0"
50
51[workspace.lints.rust]
52# `deny` (not `forbid`) so FFI workspace members (napi-rs, wasm-bindgen, PyO3) can `#[allow(unsafe_code)]` where needed.
53unsafe_code = "deny"
54missing_docs = "warn"
55
56[workspace.lints.clippy]
57all = "warn"
58pedantic = "warn"
59nursery = "warn"
60cargo = "warn"
61"#
62    .to_string()
63}

65/// README.md content
66fn readme_content() -> String {
67    r#"# {{workspace_name}}
68
69{{description}}
70
71## Architecture
72
73This workspace consists of the following crates:
74
75- **core**: Core business logic and shared functionality
76- **cli**: Command-line interface
77- **utils**: Utility functions and helpers
78
79## Building
80
81```bash
82# Build all crates
83cargo build --workspace
84
85# Run tests
86cargo test --workspace
87
88# Format code
89cargo fmt --all
90
91# Check code quality
92cargo clippy --workspace --all-features
93```
94
95## Usage
96
97```bash
98# Run the CLI application
99cargo run --bin cli
100
101# Run tests for specific crate
102cargo test -p core
103```
104
105## License
106
107This project is licensed under the MIT OR Apache-2.0 license.
108"#
109    .to_string()
110}

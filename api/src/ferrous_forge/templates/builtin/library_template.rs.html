9/// Create the library template

17/// Create the manifest for library template
18fn create_library_manifest() -> TemplateManifest {
19    let mut manifest = TemplateManifest::new("library".to_string(), TemplateKind::Library);
20
21    manifest.description = "Rust library with comprehensive testing setup".to_string();
22    manifest.author = "Ferrous Forge Team".to_string();
23
24    // Add variables
25    manifest.add_variable(TemplateVariable::required(
26        "project_name".to_string(),
27        "Name of the library".to_string(),
28    ));
29
30    manifest.add_variable(TemplateVariable::optional(
31        "author".to_string(),
32        "Author name".to_string(),
33        "Unknown".to_string(),
34    ));
35
36    // Add files
37    manifest.add_file(TemplateFile::new(
38        PathBuf::from("Cargo.toml"),
39        PathBuf::from("Cargo.toml"),
40    ));
41
42    manifest.add_file(TemplateFile::new(
43        PathBuf::from("src/lib.rs"),
44        PathBuf::from("src/lib.rs"),
45    ));
46
47    manifest.add_file(TemplateFile::new(
48        PathBuf::from(".ferrous-forge/config.toml"),
49        PathBuf::from(".ferrous-forge/config.toml"),
50    ));
51
52    // Add post-generate commands
53    manifest.add_post_generate("cargo fmt".to_string());
54    manifest.add_post_generate("cargo test".to_string());
55
56    manifest
57}

59/// Create the files for library template

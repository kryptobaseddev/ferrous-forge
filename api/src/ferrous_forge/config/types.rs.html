3use serde::{Deserialize, Serialize};
4
5/// Ferrous Forge configuration
6#[derive(Debug, Clone, Serialize, Deserialize)]
7pub struct Config {
8    /// Whether Ferrous Forge has been initialized
9    pub initialized: bool,
10    /// Version of the configuration format
11    pub version: String,
12    /// Update channel (stable, beta, nightly)
13    pub update_channel: String,
14    /// Whether to automatically check for updates
15    pub auto_update: bool,
16    /// Custom clippy rules
17    pub clippy_rules: Vec<String>,
18    /// File size limit in lines
19    pub max_file_lines: usize,
20    /// Function size limit in lines
21    pub max_function_lines: usize,
22    /// Required Rust edition (e.g. "2024") — locked; LLM agents must not change this
23    pub required_edition: String,
24    /// Required minimum rust-version (e.g. "1.85.0") — locked; LLM agents must not change this
25    pub required_rust_version: String,
26    /// Whether to ban underscore bandaid patterns
27    pub ban_underscore_bandaid: bool,
28    /// Whether to require documentation
29    pub require_documentation: bool,
30    /// Custom validation rules
31    pub custom_rules: Vec<CustomRule>,
32    /// Validation settings
33    pub validation: ValidationConfig,
34}
35
36/// Custom validation rule
37#[derive(Debug, Clone, Serialize, Deserialize)]
38pub struct CustomRule {
39    /// Name of the rule
40    pub name: String,
41    /// Pattern to match (regex)
42    pub pattern: String,
43    /// Error message to display
44    pub message: String,
45    /// Whether this rule is enabled
46    pub enabled: bool,
47}
48
49/// Validation configuration settings
50#[derive(Debug, Clone, Serialize, Deserialize)]
51pub struct ValidationConfig {
52    /// Whether to check for version consistency (`SSoT`)
53    pub check_version_consistency: Option<bool>,
54    /// Paths to exclude from version checking
55    pub version_check_exclusions: Option<Vec<String>>,
56    /// Whether to enforce Keep a Changelog format
57    pub enforce_keep_a_changelog: Option<bool>,
58    /// Whether to require changelog entry for current version
59    pub require_changelog_entry: Option<bool>,
60    /// Whether to validate changelog when creating git tags
61    pub check_changelog_on_tag: Option<bool>,
62    /// Required sections in changelog (e.g., ["Added", "Changed", "Fixed"])
63    pub changelog_required_sections: Option<Vec<String>>,
64}
65
66impl Default for ValidationConfig {
67    fn default() -> Self {
68        Self {
69            check_version_consistency: Some(true),
70            version_check_exclusions: None,
71            enforce_keep_a_changelog: Some(true),
72            require_changelog_entry: Some(true),
73            check_changelog_on_tag: Some(true),
74            changelog_required_sections: Some(vec![
75                "Added".to_string(),
76                "Changed".to_string(),
77                "Fixed".to_string(),
78            ]),
79        }
80    }
81}

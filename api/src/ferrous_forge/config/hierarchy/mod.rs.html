1//! Hierarchical configuration system
2//!
3//! Implements a three-level configuration hierarchy:
4//! 1. System: /etc/ferrous-forge/config.toml
5//! 2. User: ~/.config/ferrous-forge/config.toml  
6//! 3. Project: ./.ferrous-forge/config.toml
7
8/// Configuration level definitions (system, user, project).
9pub mod levels;
10/// Partial configuration for merging across hierarchy levels.
11pub mod partial;
12
13pub use levels::ConfigLevel;
14pub use partial::PartialConfig;
15
16use crate::config::Config;
17use crate::{Error, Result};
18use tokio::fs;
19use tracing::info;
20
21/// Hierarchical configuration manager
22pub struct HierarchicalConfig {
23    /// System-level configuration
24    pub system: Option<PartialConfig>,
25    /// User-level configuration
26    pub user: Option<PartialConfig>,
27    /// Project-level configuration
28    pub project: Option<PartialConfig>,
29}
30
31impl HierarchicalConfig {
32    /// Load configuration from all levels
33    ///
34    /// # Errors
35    ///
36    /// Returns an error if reading or parsing any configuration level fails.
37    pub async fn load() -> Result<Self> {
38        let system = PartialConfig::load_from_level(ConfigLevel::System).await?;
39        let user = PartialConfig::load_from_level(ConfigLevel::User).await?;
40        let project = PartialConfig::load_from_level(ConfigLevel::Project).await?;
41
42        Ok(Self {
43            system,
44            user,
45            project,
46        })
47    }
48
49    /// Get merged configuration with proper precedence
50    pub fn merged(&self) -> Config {
51        let mut merged = PartialConfig::default();
52
53        // Apply in order of precedence (lowest to highest)
54        if let Some(system) = &self.system {
55            merged = merged.merge(system.clone());
56        }
57        if let Some(user) = &self.user {
58            merged = merged.merge(user.clone());
59        }
60        if let Some(project) = &self.project {
61            merged = merged.merge(project.clone());
62        }
63
64        merged.to_full_config()
65    }
66
67    /// Save configuration at a specific level
68    ///
69    /// # Errors
70    ///
71    /// Returns an error if serializing the config, creating directories,
72    /// or writing the file fails.
73    pub async fn save_at_level(config: &Config, level: ConfigLevel) -> Result<()> {
74        let path = level.path()?;
75
76        // Ensure parent directory exists
77        if let Some(parent) = path.parent() {
78            fs::create_dir_all(parent).await?;
79        }
80
81        let contents = toml::to_string_pretty(config)
82            .map_err(|e| Error::config(format!("Failed to serialize config: {}", e)))?;
83
84        fs::write(&path, contents).await.map_err(|e| {
85            Error::config(format!(
86                "Failed to write {} config: {}",
87                level.display_name(),
88                e
89            ))
90        })?;
91
92        info!(
93            "Saved {} configuration to {}",
94            level.display_name(),
95            path.display()
96        );
97        Ok(())
98    }
99
100    /// Get configuration sources report
101    pub fn sources_report(&self) -> String {
102        let mut report = String::from("Configuration sources:\n");
103
104        if self.system.is_some() {
105            report.push_str(&format!("  ✓ System: /etc/ferrous-forge/config.toml\n"));
106        } else {
107            report.push_str("  ✗ System: Not found\n");
108        }
109
110        if self.user.is_some() {
111            let path = ConfigLevel::User.path().unwrap_or_default();
112            report.push_str(&format!("  ✓ User: {}\n", path.display()));
113        } else {
114            report.push_str("  ✗ User: Not found\n");
115        }
116
117        if self.project.is_some() {
118            report.push_str("  ✓ Project: ./.ferrous-forge/config.toml\n");
119        } else {
120            report.push_str("  ✗ Project: Not found\n");
121        }
122
123        report
124    }
125}

127/// Migrate from old single-file config to hierarchical system
128///
129/// # Errors
130///
131/// Returns an error if reading the old config, parsing it, saving the migrated
132/// config, or renaming the backup file fails.
133pub async fn migrate_config() -> Result<()> {
134    let old_path = Config::config_file_path()?;
135
136    if !old_path.exists() {
137        info!("No old configuration to migrate");
138        return Ok(());
139    }
140
141    info!("Migrating configuration from {}", old_path.display());
142
143    // Load old config
144    let old_contents = fs::read_to_string(&old_path)
145        .await
146        .map_err(|e| Error::config(format!("Failed to read old config: {}", e)))?;
147
148    let config: Config = toml::from_str(&old_contents)
149        .map_err(|e| Error::config(format!("Failed to parse old config: {}", e)))?;
150
151    // Save to user level (most appropriate for migrated config)
152    HierarchicalConfig::save_at_level(&config, ConfigLevel::User).await?;
153
154    // Rename old config file to backup
155    let backup_path = old_path.with_extension("toml.backup");
156    fs::rename(&old_path, &backup_path)
157        .await
158        .map_err(|e| Error::config(format!("Failed to backup old config: {}", e)))?;
159
160    info!(
161        "Migration complete. Old config backed up to {}",
162        backup_path.display()
163    );
164    Ok(())
165}

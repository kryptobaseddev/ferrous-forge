6/// Ensure rustfmt is installed and available
7pub(super) async fn ensure_rustfmt_installed() -> Result<()> {
8    let output = Command::new("rustfmt")
9        .arg("--version")
10        .output()
11        .map_err(|_| Error::tool_not_found("rustfmt"))?;
12
13    if !output.status.success() {
14        return Err(Error::tool_not_found("rustfmt"));
15    }
16
17    Ok(())
18}

29/// Analyze violations and generate a comprehensive report
30pub async fn analyze_and_generate_report(
31    project_path: &Path,
32    violations: &[Violation],
33) -> anyhow::Result<()> {
34    let analyzer = AIAnalyzer::new(project_path.to_path_buf());
35    let report = analyzer
36        .analyze_violations_async(violations.to_vec())
37        .await?;
38
39    // Save the analysis report
40    analyzer.save_analysis(&report)?;
41
42    // Generate and save orchestrator instructions
43    let instructions = generate_orchestrator_instructions(&report).await?;
44    analyzer.save_orchestrator_instructions(&instructions)?;
45
46    Ok(())
47}

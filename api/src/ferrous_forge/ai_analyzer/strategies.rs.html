7/// Generate fix strategies from violation analyses
8pub fn generate_fix_strategies(analyses: &[ViolationAnalysis]) -> Vec<FixStrategy> {
9    let mut strategies = Vec::new();
10    let mut violation_counts = std::collections::HashMap::new();
11
12    for analysis in analyses {
13        *violation_counts
14            .entry(&analysis.violation.violation_type)
15            .or_insert(0) += 1;
16    }
17
18    for (violation_type, count) in violation_counts {
19        strategies.push(create_strategy(violation_type.clone(), count));
20    }
21
22    strategies
23}
24
25fn create_strategy(violation_type: ViolationType, count: usize) -> FixStrategy {
26    match violation_type {
27        ViolationType::UnwrapInProduction => create_unwrap_strategy(violation_type, count),
28        ViolationType::UnderscoreBandaid => create_underscore_strategy(violation_type, count),
29        ViolationType::FunctionTooLarge => create_function_refactor_strategy(violation_type, count),
30        ViolationType::FileTooLarge => create_file_refactor_strategy(violation_type, count),
31        _ => create_generic_strategy(violation_type, count),
32    }
33}

35/// Create strategy for unwrap violations
36fn create_unwrap_strategy(violation_type: ViolationType, count: usize) -> FixStrategy {
37    FixStrategy {
38        violation_type,
39        strategy_name: "Replace unwrap with proper error handling".to_string(),
40        description: format!(
41            "Replace {} instances of .unwrap() with ? operator or proper error handling",
42            count
43        ),
44        implementation_steps: vec![
45            "Identify function return type".to_string(),
46            "If Result type, replace .unwrap() with ?".to_string(),
47            "If not, change return type to Result".to_string(),
48            "Update function callers".to_string(),
49        ],
50        estimated_effort: if count < 10 { "Low" } else { "Medium" }.to_string(),
51        risk_level: "Low".to_string(),
52    }
53}

134/// Create a summary of the analysis results

143/// Categorize and prioritize fixes by complexity
144fn categorize_and_prioritize_fixes(analyses: &[ViolationAnalysis]) -> Vec<String> {
145    let mut trivial_fixes = Vec::new();
146    let mut simple_fixes = Vec::new();
147    let mut moderate_fixes = Vec::new();
148
149    for analysis in analyses {
150        if !analysis.ai_fixable {
151            continue;
152        }
153
154        let fix_desc = format!(
155            "Fix {:?} at {}:{} (confidence: {:.0}%)",
156            analysis.violation.violation_type,
157            analysis.violation.file.display(),
158            analysis.violation.line,
159            analysis.confidence_score * 100.0
160        );
161
162        match analysis.fix_complexity {
163            super::types::FixComplexity::Trivial => trivial_fixes.push(fix_desc),
164            super::types::FixComplexity::Simple => simple_fixes.push(fix_desc),
165            super::types::FixComplexity::Moderate => moderate_fixes.push(fix_desc),
166            _ => {}
167        }
168    }
169
170    let mut prioritized_fixes = Vec::new();
171    prioritized_fixes.extend(trivial_fixes);
172    prioritized_fixes.extend(simple_fixes);
173    prioritized_fixes.extend(moderate_fixes);
174    
175    prioritized_fixes
176}

196/// Identify code patterns in the given content
197pub fn identify_code_patterns(content: &str) -> CodePatterns {
198    let architectural_style = detect_architectural_style(content);
199    let error_patterns = detect_error_patterns(content);
200    let common_patterns = detect_common_patterns(content);
201
202    CodePatterns {
203        architectural_style,
204        error_patterns,
205        common_patterns,
206    }
207}
208
209fn detect_architectural_style(content: &str) -> ArchitecturalStyle {
210    if content.contains("mod ") && content.matches("mod ").count() > 5 {
211        ArchitecturalStyle::Modular
212    } else if content.contains("async fn") && content.contains("tokio") {
213        ArchitecturalStyle::EventDriven
214    } else if content.contains("layer") || content.contains("Layer") {
215        ArchitecturalStyle::Layered
216    } else {
217        ArchitecturalStyle::Unknown
218    }
219}
220
221fn detect_error_patterns(content: &str) -> Vec<ErrorPattern> {
222    let mut patterns = Vec::new();
223
224    if content.contains(".unwrap()") {
225        patterns.push(ErrorPattern::UnwrapUsage);
226    }
227    if content.contains(".expect(") {
228        patterns.push(ErrorPattern::ExpectUsage);
229    }
230    if content.contains("panic!") {
231        patterns.push(ErrorPattern::PanicUsage);
232    }
233    if content.contains("let _ =") {
234        patterns.push(ErrorPattern::IgnoredError);
235    }
236    if content.contains("?") {
237        patterns.push(ErrorPattern::PropagatedError);
238    }
239
240    patterns
241}

243fn detect_common_patterns(content: &str) -> Vec<Pattern> {
244    let mut patterns = Vec::new();
245
246    let builder_count = content.matches("Builder").count();
247    if builder_count > 0 {
248        patterns.push(Pattern {
249            name: "Builder Pattern".to_string(),
250            occurrences: builder_count,
251            locations: vec![],
252        });
253    }
254
255    let factory_count = content.matches("Factory").count() + content.matches("::new(").count();
256    if factory_count > 0 {
257        patterns.push(Pattern {
258            name: "Factory Pattern".to_string(),
259            occurrences: factory_count,
260            locations: vec![],
261        });
262    }
263
264    patterns
265}

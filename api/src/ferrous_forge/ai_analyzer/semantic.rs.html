1//! Semantic analysis functionality for AI-powered violation analysis
2//!
3//! This module provides functions for analyzing the semantic context of code violations,
4//! assessing fix complexity, and performing deep code analysis to support AI-powered
5//! recommendations for resolving Ferrous Forge violations.
6
7use std::collections::HashMap;
8
9use super::types::{CodeContext, FixComplexity, SemanticAnalysis};
10use crate::validation::{Violation, ViolationType};

12/// Perform semantic analysis on a violation
13pub fn perform_semantic_analysis(
14    violation: &Violation,
15    context: &CodeContext,
16    content: &str,
17) -> SemanticAnalysis {
18    let lines: Vec<&str> = content.lines().collect();
19    let line_idx = violation.line.saturating_sub(1);
20
21    let actual_type = infer_actual_type(&lines, line_idx);
22    let expected_type = infer_expected_type(&violation.violation_type);
23    let data_flow = trace_data_flow(&lines, line_idx);
24    let control_flow = trace_control_flow(&lines, line_idx);
25    let dependencies = extract_dependencies(context);
26    let error_propagation = trace_error_propagation(&lines, line_idx);
27
28    SemanticAnalysis {
29        actual_type,
30        expected_type,
31        data_flow,
32        control_flow,
33        dependencies,
34        error_propagation,
35    }
36}
37
38fn infer_actual_type(lines: &[&str], line_idx: usize) -> Option<String> {
39    if line_idx < lines.len() {
40        let line = lines[line_idx];
41        if line.contains("Result<") {
42            Some("Result".to_string())
43        } else if line.contains("Option<") {
44            Some("Option".to_string())
45        } else if line.contains("Vec<") {
46            Some("Vec".to_string())
47        } else {
48            None
49        }
50    } else {
51        None
52    }
53}
54
55fn infer_expected_type(violation_type: &ViolationType) -> Option<String> {
56    match violation_type {
57        ViolationType::UnwrapInProduction => Some("Result or Option".to_string()),
58        ViolationType::UnderscoreBandaid
59        | ViolationType::WrongEdition
60        | ViolationType::FileTooLarge
61        | ViolationType::FunctionTooLarge
62        | ViolationType::LineTooLong
63        | ViolationType::MissingDocs
64        | ViolationType::MissingDependencies
65        | ViolationType::OldRustVersion
66        | ViolationType::LockedSetting
67        | ViolationType::MissingModuleDoc
68        | ViolationType::MissingDocConfig
69        | ViolationType::HardcodedVersion
70        | ViolationType::MissingChangelogEntry
71        | ViolationType::InvalidChangelogFormat => None,
72    }
73}

75/// Return an empty [`SemanticAnalysis`] (used for locked settings that skip content analysis)
76pub fn empty_semantic_analysis() -> SemanticAnalysis {
77    SemanticAnalysis {
78        actual_type: None,
79        expected_type: None,
80        data_flow: vec![],
81        control_flow: vec![],
82        dependencies: vec![],
83        error_propagation: vec![],
84    }

87fn trace_data_flow(lines: &[&str], line_idx: usize) -> Vec<String> {
88    let mut flow = Vec::new();
89    if line_idx < lines.len()
90        && let Some(var_name) = extract_variable_name(lines[line_idx])
91    {
92        let usage = analyze_variable_usage(lines);
93        if let Some(uses) = usage.get(&var_name) {
94            for use_line in uses {
95                flow.push(format!(
96                    "Line {}: Variable '{}' used",
97                    use_line + 1,
98                    var_name
99                ));
100            }
101        }
102    }
103    flow
104}
105
106fn trace_control_flow(lines: &[&str], line_idx: usize) -> Vec<String> {
107    let mut flow = Vec::new();
108    // Include context around the violation line
109    let start = line_idx.saturating_sub(5);
110    let end = (line_idx + 5).min(lines.len());
111
112    for (i, line) in lines.iter().enumerate().skip(start).take(end - start) {
113        if line.contains("if ") || line.contains("match ") || line.contains("while ") {
114            flow.push(format!("Line {}: Control flow statement", i + 1));
115        }
116    }
117    flow
118}
119
120fn analyze_variable_usage(lines: &[&str]) -> HashMap<String, Vec<usize>> {
121    let mut usage = HashMap::new();
122    for (i, line) in lines.iter().enumerate() {
123        if let Some(var) = extract_variable_name(line) {
124            usage.entry(var).or_insert_with(Vec::new).push(i);
125        }
126    }
127    usage
128}
129
130fn extract_variable_name(line: &str) -> Option<String> {
131    if line.contains("let ") {
132        line.split("let ")
133            .nth(1)
134            .and_then(|s| s.split('=').next())
135            .map(|s| s.trim().to_string())
136    } else {
137        None
138    }
139}
140
141fn trace_error_propagation(lines: &[&str], line_idx: usize) -> Vec<String> {
142    let mut path = Vec::new();
143    // Focus on error handling near the violation
144    let start = line_idx.saturating_sub(10);
145    let end = (line_idx + 10).min(lines.len());
146
147    for (i, line) in lines.iter().enumerate().skip(start).take(end - start) {
148        if line.contains('?') || line.contains(".unwrap()") || line.contains(".expect(") {
149            path.push(format!("Line {}: Error handling point", i + 1));
150        }
151    }
152    path
153}

155/// Assess the complexity of fixing a violation
156pub fn assess_fix_complexity(
157    violation: &Violation,
158    context: &CodeContext,
159    semantic: &SemanticAnalysis,
160) -> FixComplexity {
161    match violation.violation_type {
162        ViolationType::UnwrapInProduction => {
163            if context
164                .return_type
165                .as_ref()
166                .is_some_and(|r| r.contains("Result"))
167            {
168                if semantic.dependencies.is_empty() {
169                    FixComplexity::Trivial
170                } else {
171                    FixComplexity::Simple
172                }
173            } else {
174                FixComplexity::Moderate
175            }
176        }
177        ViolationType::UnderscoreBandaid => FixComplexity::Simple,
178        ViolationType::LineTooLong => FixComplexity::Trivial,
179        ViolationType::FunctionTooLarge => FixComplexity::Complex,
180        ViolationType::FileTooLarge => FixComplexity::Architectural,
181        ViolationType::WrongEdition
182        | ViolationType::OldRustVersion
183        | ViolationType::LockedSetting => FixComplexity::Architectural,
184        _ => FixComplexity::Moderate,
185    }
186}

188fn extract_dependencies(context: &CodeContext) -> Vec<String> {
189    let mut deps = Vec::new();
190
191    for import in &context.imports {
192        if !import.contains("std::") {
193            deps.push(import.clone());
194        }
195    }
196
197    if context.trait_impl.is_some() {
198        deps.push("Trait implementation".to_string());
199    }
200
201    if context.is_async {
202        deps.push("Async context".to_string());
203    }
204
205    if context.is_generic {
206        deps.push("Generic constraints".to_string());
207    }
208
209    deps
210}

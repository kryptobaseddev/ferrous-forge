35        println!(

37            update_info.version,

40
41        // Download the new binary

43        
44        // Verify the download if hash is provided
45        if let Some(expected_hash) = &update_info.sha256 {
46            self.verify_download(&temp_path, expected_hash).await?;

49        // Replace the current binary

58    /// Download update to temporary location
59    async fn download_update(&self, update_info: &UpdateInfo) -> Result<PathBuf> {
60        let client = reqwest::Client::new();
61        let response = client
62            .get(&update_info.download_url)
63            .send()
64            .await
65            .map_err(|e| Error::network(format!("Failed to download update: {}", e)))?;
66
67        if !response.status().is_success() {
68            return Err(Error::network(format!(
69                "Download failed: {}",
70                response.status()
71            )));
72        }
73
74        let bytes = response
75            .bytes()
76            .await
77            .map_err(|e| Error::network(format!("Failed to read download: {}", e)))?;
78
79        // Create temporary file
80        let temp_dir = std::env::temp_dir();
81        let temp_path = temp_dir.join(format!("ferrous-forge-{}", update_info.version));
82
83        fs::write(&temp_path, bytes)
84            .await
85            .map_err(|e| Error::io(format!("Failed to write temporary file: {}", e)))?;
86
87        Ok(temp_path)
88    }

90    /// Verify downloaded file hash
91    async fn verify_download(&self, path: &PathBuf, expected_hash: &str) -> Result<()> {
92        let contents = fs::read(path)
93            .await
94            .map_err(|e| Error::io(format!("Failed to read downloaded file: {}", e)))?;
95
96        use sha2::{Sha256, Digest};
97        let mut hasher = Sha256::new();
98        hasher.update(&contents);
99        let actual_hash = format!("{:x}", hasher.finalize());
100
101        if actual_hash != expected_hash {
102            return Err(Error::security(format!(
103                "Hash verification failed. Expected: {}, Got: {}",
104                expected_hash, actual_hash
105            )));
106        }
107
108        Ok(())
109    }

111    /// Replace current binary with updated version
112    async fn replace_binary(&self, temp_path: &PathBuf) -> Result<()> {
113        // Create backup of current binary
114        let backup_path = self.binary_path.with_extension("backup");
115        fs::copy(&self.binary_path, &backup_path)
116            .await
117            .map_err(|e| Error::io(format!("Failed to create backup: {}", e)))?;
118
119        // Replace with new binary
120        fs::copy(temp_path, &self.binary_path)
121            .await
122            .map_err(|e| {
123                // Try to restore backup on failure
124                // Attempt to restore backup, ignore errors since we're already in error state
125                drop(std::fs::copy(&backup_path, &self.binary_path));
126                Error::io(format!("Failed to replace binary: {}", e))
127            })?;
128
129        // Make executable on Unix systems
130        #[cfg(unix)]
131        {
132            use std::os::unix::fs::PermissionsExt;
133            let mut perms = fs::metadata(&self.binary_path)
134                .await
135                .map_err(|e| Error::io(format!("Failed to get file metadata: {}", e)))?
136                .permissions();
137            perms.set_mode(0o755);
138            fs::set_permissions(&self.binary_path, perms)
139                .await
140                .map_err(|e| Error::io(format!("Failed to set permissions: {}", e)))?;
141        }
142
143        // Clean up temporary file and backup
144        // Clean up temporary files, ignore errors
145        drop(fs::remove_file(temp_path).await);
146        drop(fs::remove_file(&backup_path).await);
147
148        Ok(())
149    }

178/// Format file size in human-readable format
179fn format_file_size(bytes: u64) -> String {
180    const UNITS: &[&str] = &["B", "KB", "MB", "GB"];
181    const THRESHOLD: u64 = 1024;
182
183    if bytes < THRESHOLD {
184        return format!("{} B", bytes);
185    }
186
187    let mut size = bytes as f64;
188    let mut unit_index = 0;
189
190    while size >= THRESHOLD as f64 && unit_index < UNITS.len() - 1 {
191        size /= THRESHOLD as f64;
192        unit_index += 1;
193    }
194
195    format!("{:.1} {}", size, UNITS[unit_index])
196}

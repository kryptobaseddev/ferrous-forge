299fn filter_violations(
300    violations: &[Violation],
301    only_types: &Option<HashSet<String>>,
302    skip_types: &Option<HashSet<String>>,
303    limit: Option<usize>,
304) -> Vec<Violation> {
305    let mut filtered: Vec<Violation> = violations
306        .iter()
307        .filter(|v| {
308            let violation_type = format!("{:?}", v.violation_type).to_uppercase();
309
310            // Check only filter
311            if let Some(only) = only_types {
312                if !only.contains(&violation_type) {
313                    return false;
314                }
315            }
316
317            // Check skip filter
318            if let Some(skip) = skip_types {
319                if skip.contains(&violation_type) {
320                    return false;
321                }
322            }
323
324            // Only include violations we can potentially auto-fix
325            can_potentially_auto_fix(v)
326        })
327        .cloned()
328        .collect();
329
330    // Apply limit if specified
331    if let Some(limit) = limit {
332        filtered.truncate(limit);
333    }
334
335    filtered
336}

338fn can_potentially_auto_fix(violation: &Violation) -> bool {
339    match violation.violation_type {
340        ViolationType::UnwrapInProduction => true,
341        ViolationType::UnderscoreBandaid => true,
342        ViolationType::LineTooLong => false, // Requires intelligent reformatting
343        ViolationType::FunctionTooLarge => false, // Requires refactoring
344        ViolationType::FileTooLarge => false, // Requires architectural changes
345        _ => false,
346    }
347}

349fn group_violations_by_file(violations: &[Violation]) -> HashMap<PathBuf, Vec<Violation>> {
350    let mut grouped: HashMap<PathBuf, Vec<Violation>> = HashMap::new();
351
352    for violation in violations {
353        grouped
354            .entry(violation.file.clone())
355            .or_default()
356            .push(violation.clone());
357    }
358
359    // Sort violations within each file by line number (reverse order for safe fixing)
360    for file_violations in grouped.values_mut() {
361        file_violations.sort_by(|a, b| b.line.cmp(&a.line));
362    }
363
364    grouped
365}

367fn fix_file_violations(
368    file_path: &Path,
369    violations: &[Violation],
370    dry_run: bool,
371) -> anyhow::Result<(usize, usize)> {
372    let content = fs::read_to_string(file_path)
373        .with_context(|| format!("Failed to read file: {}", file_path.display()))?;
374
375    let mut lines: Vec<String> = content.lines().map(|l| l.to_string()).collect();
376    let mut fixed_count = 0;
377    let mut skipped_count = 0;
378
379    // Analyze file context for safer fixes
380    let file_context = analyze_file_context(&content);
381
382    for violation in violations {
383        if violation.line == 0 || violation.line > lines.len() {
384            continue;
385        }
386
387        let line_idx = violation.line - 1;
388        let original_line = lines[line_idx].clone();
389
390        match fix_violation_in_line(&original_line, violation, &file_context) {
391            FixResult::Fixed(fixed_line) => {
392                if !dry_run {
393                    lines[line_idx] = fixed_line;
394                }
395                fixed_count += 1;
396            }
397            FixResult::Skipped(_reason) => {
398                skipped_count += 1;
399            }
400            FixResult::NotApplicable => {}
401        }
402    }
403
404    if fixed_count > 0 && !dry_run {
405        let fixed_content = lines.join("\n");
406        // Preserve final newline if it existed
407        let fixed_content = if content.ends_with('\n') {
408            format!("{}\n", fixed_content)
409        } else {
410            fixed_content
411        };
412        fs::write(file_path, fixed_content)
413            .with_context(|| format!("Failed to write file: {}", file_path.display()))?;
414    }
415
416    Ok((fixed_count, skipped_count))
417}

419/// Context about a file that helps make safer fix decisions
420#[derive(Debug)]
421struct FileContext {
422    is_test_file: bool,
423    is_example_file: bool,
424    is_bin_file: bool,
425    has_anyhow_import: bool,
426    has_thiserror_import: bool,
427    function_signatures: Vec<FunctionSignature>,
428}

430#[derive(Debug)]
431struct FunctionSignature {
432    #[allow(dead_code)]
433    line: usize,
434    #[allow(dead_code)]
435    name: String,
436    returns_result: bool,
437    returns_option: bool,
438    is_test: bool,
439    is_main: bool,
440}

442fn analyze_file_context(content: &str) -> FileContext {
443    let lines: Vec<&str> = content.lines().collect();
444
445    // Check file type based on common patterns
446    let is_test_file = content.contains("#[cfg(test)]") || content.contains("#[test]");
447    let is_example_file = content.contains("fn main()") && content.lines().count() < 100;
448    let is_bin_file = content.contains("fn main()");
449
450    // Check for error handling libraries
451    let has_anyhow_import = content.contains("use anyhow::") || content.contains("anyhow::");
452    let has_thiserror_import =
453        content.contains("use thiserror::") || content.contains("thiserror::");
454
455    // Analyze function signatures - handle multi-line signatures
456    let mut function_signatures = Vec::new();
457    let mut i = 0;
458    while i < lines.len() {
459        if let Some(sig) = parse_function_signature_multiline(&lines, i) {
460            function_signatures.push(sig);
461            // Skip to the end of this function signature
462            while i < lines.len() && !lines[i].contains('{') {
463                i += 1;
464            }
465        }
466        i += 1;
467    }
468
469    FileContext {
470        is_test_file,
471        is_example_file,
472        is_bin_file,
473        has_anyhow_import,
474        has_thiserror_import,
475        function_signatures,
476    }
477}

479fn parse_function_signature_multiline(
480    lines: &[&str], 
481    start_idx: usize
482) -> Option<FunctionSignature> {
483    let start_line = lines.get(start_idx)?;
484    let trimmed = start_line.trim();
485
486    // Check if this is a function declaration
487    if !trimmed.starts_with("fn ")
488        && !trimmed.starts_with("pub fn ")
489        && !trimmed.starts_with("async fn ")
490        && !trimmed.starts_with("pub async fn ")
491    {
492        return None;
493    }
494
495    // Extract function name
496    let fn_start = trimmed.find("fn ")? + 3;
497    let fn_end = trimmed[fn_start..].find('(')?;
498    let name = trimmed[fn_start..fn_start + fn_end].trim().to_string();
499
500    // Collect the full signature (might span multiple lines)
501    let mut full_signature = String::new();
502    let mut idx = start_idx;
503    let mut found_open_brace = false;
504    
505    // Check if there's a #[test] attribute above
506    let is_test = if start_idx > 0 {
507        lines[start_idx - 1].contains("#[test]")
508    } else {
509        false
510    };
511    
512    while idx < lines.len() && !found_open_brace {
513        full_signature.push_str(lines[idx]);
514        full_signature.push(' ');
515        if lines[idx].contains('{') {
516            found_open_brace = true;
517        }
518        idx += 1;
519    }
520
521    // Check return type in the full signature
522    let returns_result = full_signature.contains("-> Result") 
523        || full_signature.contains("-> anyhow::Result")
524        || full_signature.contains("-> std::result::Result");
525    let returns_option = full_signature.contains("-> Option");
526    let is_main = name == "main";
527
528    Some(FunctionSignature {
529        line: start_idx,
530        name,
531        returns_result,
532        returns_option,
533        is_test,
534        is_main,
535    })
536}

538enum FixResult {

540    Skipped(String),
541    NotApplicable,

544fn fix_violation_in_line(line: &str, violation: &Violation, context: &FileContext) -> FixResult {
545    match violation.violation_type {
546        ViolationType::UnwrapInProduction => fix_unwrap_in_line(line, violation, context),
547        ViolationType::UnderscoreBandaid => fix_underscore_in_line(line, violation, context),
548        _ => FixResult::NotApplicable,
549    }
550}

552fn fix_unwrap_in_line(line: &str, _violation: &Violation, context: &FileContext) -> FixResult {
553    // Skip fixes in test code
554    if context.is_test_file {
555        return FixResult::Skipped("Test file - manual review needed".to_string());
556    }
557
558    if line.contains(".unwrap()") {
559        // Don't fix if it's in a string literal
560        if line.contains("\".unwrap()\"") || line.contains(r#"'.unwrap()'"#) {
561            return FixResult::Skipped("String literal, not actual code".to_string());
562        }
563        
564        // Check if we're in a function that can use ?
565        let can_use_question_mark = check_can_use_question_mark(line, context);
566
567        if can_use_question_mark {
568            // Safe to replace with ?
569            return FixResult::Fixed(line.replace(".unwrap()", "?"));
570        } else {
571            // For main functions or examples, use expect
572            if context.is_bin_file || context.is_example_file {
573                let fixed = line.replace(".unwrap()", r#".expect("Failed to complete operation")"#);
574                return FixResult::Fixed(fixed);
575            }
576            // For other cases where we can't use ?, suggest manual review
577            return FixResult::Skipped("Cannot use ? - needs manual review".to_string());
578        }
579    }
580
581    if line.contains(".expect(") {
582        // Don't fix if it's in a string literal
583        if line.contains("\".expect(") || line.contains("'.expect(") {
584            return FixResult::Skipped("String literal, not actual code".to_string());
585        }
586        
587        // Check if we can use ?
588        let can_use_question_mark = check_can_use_question_mark(line, context);
589        
590        if can_use_question_mark {
591            // If we have anyhow, use .context()
592            if context.has_anyhow_import {
593                // Extract the expect message and convert to context
594                if let Some(start) = line.find(".expect(\"") {
595                    if let Some(end) = line[start + 9..].find("\")") {
596                        let message = &line[start + 9..start + 9 + end];
597                        let before = &line[..start];
598                        let after = &line[start + 9 + end + 2..];
599
600                        let fixed = format!("{}.context(\"{}\")?{}", before, message, after);
601                        return FixResult::Fixed(fixed);
602                    }
603                }
604            } else {
605                // Without anyhow, just replace with ?
606                // Find the expect call and replace it
607                if let Some(start) = line.find(".expect(") {
608                    // Find the matching closing paren
609                    let rest = &line[start + 8..];
610                    let mut depth = 1;
611                    let mut end_pos = 0;
612                    for (i, ch) in rest.chars().enumerate() {
613                        match ch {
614                            '(' => depth += 1,
615                            ')' => {
616                                depth -= 1;
617                                if depth == 0 {
618                                    end_pos = i;
619                                    break;
620                                }
621                            }
622                            _ => {}
623                        }
624                    }
625                    if depth == 0 {
626                        let before = &line[..start];
627                        let after = &line[start + 8 + end_pos + 1..];
628                        let fixed = format!("{}?{}", before, after);
629                        return FixResult::Fixed(fixed);
630                    }
631                }
632            }
633        }
634    }
635
636    FixResult::NotApplicable
637}

639fn check_can_use_question_mark(_line: &str, context: &FileContext) -> bool {
640    // In test files, don't use ? in tests
641    if context.is_test_file {
642        return false;
643    }
644
645    // Check if ANY function in the file returns Result or Option
646    // This is still conservative but better than before
647    let has_result_returning_functions = context.function_signatures.iter()
648        .any(|sig| !sig.is_test && !sig.is_main && (sig.returns_result || sig.returns_option));
649    
650    if has_result_returning_functions {
651        return true;
652    }
653    
654    // Check if main returns Result (common pattern)
655    let main_returns_result = context.function_signatures.iter()
656        .any(|sig| sig.is_main && sig.returns_result);
657        
658    if main_returns_result {
659        return true;
660    }
661
662    // If we have error handling imports, we likely have Result-returning functions
663    context.has_anyhow_import || context.has_thiserror_import
664}

666fn fix_underscore_in_line(line: &str, _violation: &Violation, context: &FileContext) -> FixResult {
667    // Fix underscore parameters in function signatures
668    if line.contains("fn ") && line.contains('_') {
669        // Check if this is a trait implementation where we can't change signature
670        if line.contains("impl") || line.contains("trait") {
671            return FixResult::Skipped("Cannot modify trait method signatures".to_string());
672        }
673
674        // For now, just warn about underscore parameters
675        // Proper fix would require analyzing if the parameter is actually needed
676        return FixResult::Skipped("Underscore parameters need manual review".to_string());
677    }
678
679    // Fix let _ = assignments
680    if line.trim_start().starts_with("let _ =") {
681        // Check if the expression likely returns Result
682        if let Some(rest) = line.trim_start().strip_prefix("let _ =") {
683            let rest = rest.trim();
684
685            // Don't fix if it's a drop pattern
686            if rest.contains("drop(") || rest.contains("mem::drop") {
687                return FixResult::Skipped("Intentional drop pattern".to_string());
688            }
689
690            // Don't fix if it's likely not returning Result
691            if rest.contains(".clone()") || rest.contains(".to_string()") {
692                return FixResult::Skipped("Expression likely doesn't return Result".to_string());
693            }
694
695            if let Some(expr) = rest.strip_suffix(';') {
696                let indent = &line[..line.find("let").unwrap_or(0)];
697
698                // Check if we can use ?
699                if check_can_use_question_mark(line, context) {
700                    let fixed = format!("{}{}?;", indent, expr);
701                    return FixResult::Fixed(fixed);
702                } else {
703                    // Use explicit error handling
704                    let fixed = format!(
705                        "{}if let Err(e) = {} {{ eprintln!(\"Error: {{}}\", e); }}",
706                        indent, expr
707                    );
708                    return FixResult::Fixed(fixed);
709                }
710            }
711        }
712    }
713
714    FixResult::NotApplicable
715}

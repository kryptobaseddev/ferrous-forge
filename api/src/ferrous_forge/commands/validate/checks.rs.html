36/// Check code formatting
37async fn check_code_formatting(project_path: &PathBuf) {
38    println!("üìù Checking code formatting...");
39    
40    match formatting::check_formatting(project_path).await {
41        Ok(format_result) => {
42            println!("{}", format_result.report());
43            if !format_result.unformatted_files.is_empty() {
44                println!("Files needing formatting:");
45                for file in &format_result.unformatted_files {
46                    println!("  {}", file);
47                }
48            }
49        }
50        Err(e) => {
51            println!("‚ùå Code formatting check failed: {}", e);
52        }
53    }
54    println!();

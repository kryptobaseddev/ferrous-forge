9/// Run all additional checks (documentation, formatting, security, test coverage)
10pub async fn run_additional_checks(project_path: &Path) {
11    check_documentation_coverage(project_path).await;
12    check_code_formatting(project_path).await;
13    run_security_audit_check(project_path).await;
14    check_test_coverage(project_path).await;
15}

17/// Check documentation coverage
18async fn check_documentation_coverage(project_path: &Path) {
19    println!("📚 Checking documentation coverage...");
20
21    // Add small delay to ensure proper output ordering
22    tokio::time::sleep(tokio::time::Duration::from_millis(50)).await;
23
24    match doc_coverage::check_documentation_coverage(project_path).await {
25        Ok(coverage) => {
26            println!("{}", coverage.report());
27            if coverage.coverage_percent < 80.0 {
28                println!("⚠️  Documentation coverage below 80%");
29            }
30        }
31        Err(e) => {
32            println!("❌ Documentation coverage check failed: {}", e);
33        }
34    }
35    println!();
36}

38/// Check code formatting
39async fn check_code_formatting(project_path: &Path) {
40    println!("📝 Checking code formatting...");
41
42    match formatting::check_formatting(project_path).await {
43        Ok(format_result) => {
44            println!("{}", format_result.report());
45            if !format_result.unformatted_files.is_empty() {
46                println!("Files needing formatting:");
47                for file in &format_result.unformatted_files {
48                    println!("  {}", file);
49                }
50            }
51        }
52        Err(e) => {
53            println!("❌ Code formatting check failed: {}", e);
54        }
55    }
56    println!();
57}

73/// Check test coverage
74async fn check_test_coverage(project_path: &Path) {
75    println!("🧪 Checking test coverage...");
76
77    let analyzer = CoverageAnalyzer::new();
78
79    // First check if tarpaulin is installed
80    match analyzer.check_tarpaulin_installed() {
81        Ok(true) => {
82            // Tarpaulin is installed, run coverage analysis
83            match analyzer.run_coverage(project_path).await {
84                Ok(report) => {
85                    println!("✅ Test coverage: {:.1}%", report.line_coverage);
86                    println!(
87                        "   Lines tested: {}/{}",
88                        report.lines_tested, report.total_lines
89                    );
90                    println!(
91                        "   Functions tested: {}/{}",
92                        report.functions_tested, report.total_functions
93                    );
94
95                    // Check against minimum threshold (configurable, defaulting to 80%)
96                    let threshold = 80.0;
97                    if let Err(e) = analyzer.enforce_minimum_coverage(&report, threshold) {
98                        println!("⚠️  {}", e);
99                    }
100                }
101                Err(e) => {
102                    println!("⚠️  Test coverage check failed: {}", e);
103                    println!("   This may be due to test failures or configuration issues");
104                }
105            }
106        }
107        Ok(false) => {
108            println!("⚠️  cargo-tarpaulin not installed");
109            println!("   Install with: cargo install cargo-tarpaulin");
110            println!("   Skipping test coverage check");
111        }
112        Err(e) => {
113            println!("⚠️  Failed to check for cargo-tarpaulin: {}", e);
114        }
115    }
116
117    println!();
118}

120/// Run clippy validation
121///
122/// # Errors
123///
124/// Returns an error if clippy fails to execute.
125pub async fn run_clippy_validation(
126    validator: &RustValidator,
127) -> Result<crate::validation::ClippyResult> {
128    println!("🔧 Running Clippy with strict configuration...");
129
130    let clippy_result = validator.run_clippy().await?;
131
132    if !clippy_result.success {
133        println!("❌ Clippy found issues:");
134        println!("{}", clippy_result.output);
135    } else {
136        println!("✅ Clippy validation passed!");
137    }
138
139    Ok(clippy_result)
140}

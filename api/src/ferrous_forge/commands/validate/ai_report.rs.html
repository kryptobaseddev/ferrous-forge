65/// Generate AI-friendly compliance report
66pub async fn generate_ai_report(project_path: &PathBuf, violations: &[Violation]) -> Result<()> {
67    let timestamp = chrono::Utc::now();
68    let reports_dir = setup_reports_directory(project_path).await?;
69    let timestamp_str = timestamp.format("%Y%m%d_%H%M%S").to_string();
70
71    let violation_counts = utils::count_violations_by_type(violations);
72    let ai_violations = create_ai_violations(violations).await;
73    let fix_instructions = generate_fix_instructions(violation_counts);
74    let compliance = utils::calculate_compliance(project_path, violations).await?;
75
76    let report = build_ai_report(
77        project_path,
78        &timestamp,
79        violations.len(),
80        ai_violations,
81        fix_instructions,
82        compliance,
83    );
84
85    save_and_link_reports(&reports_dir, &timestamp_str, &report).await?;
86    print_report_summary(&reports_dir, &timestamp_str);
87
88    Ok(())
89}

91async fn setup_reports_directory(project_path: &PathBuf) -> Result<PathBuf> {

93    fs::create_dir_all(&reports_dir).await?;
94    Ok(reports_dir)

98async fn create_ai_violations(violations: &[Violation]) -> Vec<AIViolation> {
99    let mut ai_violations = Vec::new();
100    for violation in violations {
101        let code_snippet = utils::get_code_snippet(&violation.file, violation.line)
102            .await
103            .unwrap_or_else(|_| "Could not read file".to_string());
104
105        let (suggested_fix, auto_fixable, priority) = 
106            get_fix_suggestion(&violation.violation_type, &violation.message);
107
108        ai_violations.push(AIViolation {
109            violation_type: format!("{:?}", violation.violation_type),
110            file: violation.file.display().to_string(),
111            line: violation.line,
112            message: violation.message.clone(),
113            code_snippet,
114            suggested_fix,
115            auto_fixable,
116            priority,
117        });
118    }
119    ai_violations
120}

122fn get_fix_suggestion(
123    violation_type: &crate::validation::ViolationType,
124    _message: &str,
125) -> (String, bool, u8) {
126    match violation_type {
127        crate::validation::ViolationType::UnderscoreBandaid => (
128            "Remove unused parameter or implement missing functionality".to_string(),
129            true,
130            2,
131        ),
132        crate::validation::ViolationType::UnwrapInProduction => (
133            "Replace `.unwrap()` with `?` or proper error handling".to_string(),
134            true,
135            1,
136        ),
137        crate::validation::ViolationType::FileTooLarge => (
138            "Split file into smaller modules following single responsibility principle".to_string(),
139            false,
140            3,
141        ),
142        crate::validation::ViolationType::FunctionTooLarge => (
143            "Extract helper functions or split into smaller, focused functions".to_string(),
144            false,
145            3,
146        ),
147        _ => (
148            "Review and fix according to Ferrous Forge standards".to_string(),
149            false,
150            4,
151        ),
152    }
153}

155fn generate_fix_instructions(violation_counts: HashMap<String, usize>) -> Vec<FixInstruction> {
156    let mut fix_instructions = Vec::new();
157    for (vtype, count) in violation_counts {
158        let (strategy, example, effort) = get_fix_strategy(&vtype);
159        fix_instructions.push(FixInstruction {
160            violation_type: vtype,
161            count,
162            fix_strategy: strategy,
163            example_fix: example,
164            effort_level: effort,
165        });
166    }
167    fix_instructions
168}

170fn get_fix_strategy(vtype: &str) -> (String, String, String) {
171    match vtype {
172        "UnderscoreBandaid" => (
173            "1. Identify what functionality the parameter should provide\n\
174            2. Either implement the functionality or remove the parameter\n\
175            3. Update function signature and callers".to_string(),
176            "// Before: fn process(_unused: String, data: Data)\n\
177            // After: fn process(data: Data) or implement the unused parameter".to_string(),
178            "Moderate".to_string(),
179        ),
180        "UnwrapInProduction" => (
181            "1. Change function to return Result<T, Error>\n\
182            2. Replace unwrap with ?\n3. Handle errors at call sites".to_string(),
183            "// Before: value.unwrap()\n// After: value?".to_string(),
184            "Easy".to_string(),
185        ),
186        "FileTooLarge" => (
187            "1. Identify logical boundaries\n2. Extract modules\n\
188            3. Move related functions to new modules\n\
189            4. Update imports".to_string(),
190            "// Extract to separate modules like: \n\
191            validation/core.rs, validation/types.rs".to_string(),
192            "Hard".to_string(),
193        ),
194        _ => (
195            "Review and fix manually".to_string(),
196            "".to_string(),
197            "Moderate".to_string(),
198        ),
199    }
200}

203fn build_ai_report(
204    project_path: &PathBuf,
205    timestamp: &chrono::DateTime<chrono::Utc>,
206    total_violations: usize,
207    ai_violations: Vec<AIViolation>,
208    fix_instructions: Vec<FixInstruction>,
209    compliance_percentage: f64,
210) -> AIReport {
211    AIReport {
212        metadata: AIMetadata {
213            timestamp: timestamp.to_rfc3339(),
214            project_path: project_path.display().to_string(),
215            ferrous_forge_version: env!("CARGO_PKG_VERSION").to_string(),
216            total_violations,
217            report_version: "1.0.0".to_string(),
218        },
219        summary: AISummary {
220            compliance_percentage,
221            files_analyzed: 0, // TODO: pass this properly
222            most_critical_issues: vec![
223                "UnderscoreBandaid violations (implement missing functionality)".to_string(),
224                "Large files need splitting".to_string(),
225                "Large functions need refactoring".to_string(),
226            ],
227            estimated_fix_time_hours: total_violations as f64 * 0.25,
228        },
229        violations: ai_violations,
230        fix_instructions,
231    }
232}

234async fn save_and_link_reports(
235    reports_dir: &PathBuf,
236    timestamp_str: &str,
237    report: &AIReport,
238) -> Result<()> {
239    let json_path = reports_dir.join(format!("ai_compliance_{}.json", timestamp_str));
240    let json_content = serde_json::to_string_pretty(&report)
241        .map_err(|e| crate::Error::config(format!("Failed to serialize AI report: {}", e)))?;
242    fs::write(&json_path, json_content).await?;
243
244    let md_path = reports_dir.join(format!("ai_compliance_{}.md", timestamp_str));
245    let md_content = generate_markdown_report(&report);
246    fs::write(&md_path, md_content).await?;
247
248    let latest_json = reports_dir.join("latest_ai_report.json");
249    let latest_md = reports_dir.join("latest_ai_report.md");
250    fs::copy(&json_path, &latest_json).await?;
251    fs::copy(&md_path, &latest_md).await?;
252
253    Ok(())
254}

13/// Execute the main fix process
14pub async fn execute_fix_process(
15    _project_path: &Path,
16    dry_run: bool,
17    filter_options: FilterOptions,
18    ai_mode: bool,
19) -> Result<()> {
20    // Validate project and get violations
21    let violations = validate_project(_project_path).await?;
22    
23    // Filter violations based on user preferences
24    let filtered_violations = validate_and_filter_violations(
25        violations,
26        &filter_options,
27        _project_path
28    ).await?;
29
30    if filtered_violations.is_empty() {
31        println!("‚úÖ No violations found that can be auto-fixed!");
32        return Ok(());
33    }
34
35    print_violations_summary(&filtered_violations);
36
37    // Run AI analysis if requested
38    if ai_mode {
39        run_ai_analysis(_project_path, &filtered_violations).await;
40    }
41
42    if !dry_run {
43        print_warning_banner();
44    }
45
46    // Group violations by file for processing
47    let violations_by_file = group_violations_by_file(&filtered_violations);
48    
49    // Process all files
50    let stats = process_all_files(violations_by_file, dry_run);
51    
52    print_final_summary(stats, dry_run);
53    
54    Ok(())
55}

57/// Validate project and return violations
58async fn validate_project(project_path: &Path) -> Result<Vec<Violation>> {
59    let validator = RustValidator::new(project_path.to_path_buf())?;
60    validator.validate_project().await
61}

63/// Validate and filter violations based on fix capabilities
64async fn validate_and_filter_violations(
65    violations: Vec<Violation>,
66    filter_options: &FilterOptions,
67    _project_path: &Path,
68) -> Result<Vec<Violation>> {
69    // Filter violations that we can potentially fix
70    let fixable_violations = filter_violations(
71        &violations,
72        &filter_options.only_types,
73        &filter_options.skip_types,
74        None // limit
75    );
76    
77    if fixable_violations.is_empty() {
78        println!("‚ÑπÔ∏è  No fixable violations found with current filters.");
79        return Ok(vec![]);
80    }
81
82    // For now, we only support specific violation types
83    let supported_types = HashSet::from([
84        "LINETOOLONG",
85        "UNDERSCORE_BANDAID", 
86        "UNWRAP_IN_PRODUCTION",
87    ]);
88
89    let filtered: Vec<_> = fixable_violations
90        .into_iter()
91        .filter(|v| {
92            let violation_str = format!("{:?}", v.violation_type).to_uppercase();
93            supported_types.contains(violation_str.as_str())
94        })
95        .collect();
96
97    if filtered.is_empty() {
98        println!("‚ÑπÔ∏è  No violations of supported types found for auto-fixing.");
99        println!("    Supported types: LINETOOLONG, UNDERSCORE_BANDAID, UNWRAP_IN_PRODUCTION");
100    }
101
102    Ok(filtered)
103}

105/// Print summary of violations to be fixed
106fn print_violations_summary(violations: &[Violation]) {
107    println!();
108    println!("üìã Found {} violations that can be auto-fixed:", violations.len());
109    
110    let mut counts = std::collections::HashMap::new();
111    for violation in violations {
112        *counts.entry(&violation.violation_type).or_insert(0) += 1;
113    }
114    
115    for (vtype, count) in counts {
116        println!("   ‚Ä¢ {:?}: {} violations", vtype, count);
117    }
118    println!();
119}

157/// Print final summary of fix results

166        println!();
167        println!("üí° Run without --dry-run to apply these fixes");
168    } else {
169        println!("   {} violations fixed", stats.total_fixed);
170        println!("   {} violations skipped", stats.total_skipped);
171        println!("   {} files modified", stats.files_modified);
172        
173        if stats.total_fixed > 0 {
174            println!();
175            println!("üîÑ Remember to run your tests to ensure everything still works!");
176        }

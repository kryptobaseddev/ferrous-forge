33/// Execute the fix command with optional AI analysis
34pub async fn execute_with_ai(
35    path: Option<PathBuf>,
36    only: Option<String>,
37    skip: Option<String>,
38    dry_run: bool,
39    _limit: Option<usize>,
40    ai_analysis: bool,
41) -> Result<()> {
42    let project_path = path.unwrap_or_else(|| std::env::current_dir().unwrap_or_default());
43
44    print_startup_banner(&project_path, dry_run);
45
46    let filter_options = parse_filter_options(only, skip);
47    execute_fix_process(&project_path, dry_run, filter_options, ai_analysis).await
48}

66/// Parse filter options from command line arguments
67fn parse_filter_options(only: Option<String>, skip: Option<String>) -> FilterOptions {
68    let only_types: Option<HashSet<String>> = only
69        .as_ref()
70        .map(|s| s.split(',').map(|t| t.trim().to_uppercase()).collect());
71
72    let skip_types: Option<HashSet<String>> = skip
73        .as_ref()
74        .map(|s| s.split(',').map(|t| t.trim().to_uppercase()).collect());
75
76    FilterOptions {
77        only_types,
78        skip_types,
79    }
80}

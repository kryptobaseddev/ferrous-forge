9/// Create a new project from template
10///
11/// # Errors
12///
13/// Returns an error if the template is not found, variables are invalid,
14/// temporary files cannot be created, or the project generation fails.
15pub async fn create_from_template(
16    template_name: &str,
17    output_dir: &Path,
18    variables: &[String],
19) -> Result<()> {
20    display_creation_header(template_name);
21
22    let registry = TemplateRegistry::new();
23    let template = get_template(&registry, template_name)?;
24
25    // Setup template files in temporary directory
26    let template_dir = setup_template_files(&template)?;
27
28    // Parse and collect variables
29    let mut template_vars = collect_template_variables(variables)?;
30
31    // Prompt for any missing required variables
32    prompt_for_missing_variables(&template, &mut template_vars)?;
33
34    // Create the project using the engine
35    let mut engine = TemplateEngine::new(template.manifest.clone(), template_dir.clone());
36
37    // Set variables
38    for (key, value) in template_vars {
39        engine.set_variable(key, value)?;
40    }
41
42    // Generate the project
43    engine.generate(output_dir)?;
44
45    display_creation_success(output_dir);
46    Ok(())
47}

49fn get_template<'a>(
50    registry: &'a TemplateRegistry,
51    template_name: &str,
52) -> Result<&'a crate::templates::registry::BuiltinTemplate> {
53    registry
54        .get_builtin(template_name)
55        .ok_or_else(|| Error::template(format!("Template '{}' not found", template_name)))
56}

71fn setup_template_files(template: &crate::templates::registry::BuiltinTemplate) -> Result<PathBuf> {
72    use std::io::Write;
73    use tempfile::TempDir;
74
75    let temp_dir = TempDir::new()
76        .map_err(|e| Error::template(format!("Failed to create temporary directory: {}", e)))?;
77
78    let template_path = temp_dir.path().join("template");
79    std::fs::create_dir_all(&template_path)
80        .map_err(|e| Error::template(format!("Failed to create template directory: {}", e)))?;
81
82    // Write manifest
83    let manifest_path = template_path.join("template.toml");
84    let manifest_toml = toml::to_string(&template.manifest)
85        .map_err(|e| Error::template(format!("Failed to serialize manifest: {}", e)))?;
86    let mut manifest_file = std::fs::File::create(&manifest_path)
87        .map_err(|e| Error::template(format!("Failed to create manifest file: {}", e)))?;
88    manifest_file
89        .write_all(manifest_toml.as_bytes())
90        .map_err(|e| Error::template(format!("Failed to write manifest: {}", e)))?;
91
92    // Write template files
93    for (file_path, content) in &template.files {
94        let full_path = template_path.join(file_path);
95        if let Some(parent) = full_path.parent() {
96            std::fs::create_dir_all(parent)
97                .map_err(|e| Error::template(format!("Failed to create directory: {}", e)))?;
98        }
99
100        let mut file = std::fs::File::create(&full_path)
101            .map_err(|e| Error::template(format!("Failed to create file: {}", e)))?;
102        file.write_all(content.as_bytes())
103            .map_err(|e| Error::template(format!("Failed to write file: {}", e)))?;
104    }
105
106    // Prevent cleanup by returning owned path
107    Ok(template_path)
108}

110fn collect_template_variables(variables: &[String]) -> Result<HashMap<String, String>> {
111    let mut template_vars = HashMap::new();
112
113    for var in variables {
114        let (key, value) = parse_var(var)?;
115        template_vars.insert(key, value);
116    }
117
118    Ok(template_vars)
119}
120
121fn parse_var(s: &str) -> Result<(String, String)> {
122    let parts: Vec<&str> = s.splitn(2, '=').collect();
123    if parts.len() != 2 {
124        return Err(Error::template(format!(
125            "Invalid variable format: '{}'. Use key=value",
126            s
127        )));
128    }
129    Ok((parts[0].to_string(), parts[1].to_string()))
130}
131
132fn prompt_for_missing_variables(
133    template: &crate::templates::registry::BuiltinTemplate,
134    template_vars: &mut HashMap<String, String>,
135) -> Result<()> {
136    use dialoguer::Input;
137
138    for var in &template.manifest.variables {
139        if var.required && !template_vars.contains_key(&var.name) {
140            let prompt = format!("{} ({})", var.name, var.description);
141
142            let value: String = Input::new()
143                .with_prompt(&prompt)
144                .interact()
145                .map_err(|e| Error::template(format!("Failed to get input: {}", e)))?;
146
147            template_vars.insert(var.name.clone(), value);
148        }
149    }
150
151    Ok(())
152}

14impl SafetyCheck for LicenseCheck {
15    async fn run(project_path: &Path) -> Result<CheckResult> {
16        run(project_path).await
17    }
18    
19    fn name() -> &'static str {
20        "license"
21    }
22    
23    fn description() -> &'static str {
24        "Validates license compatibility and presence"
25    }

28/// Validate license configuration
29pub async fn run(project_path: &Path) -> Result<CheckResult> {
30    let start = Instant::now();
31    let mut result = CheckResult::new(CheckType::License);
32    
33    // Check Cargo.toml for license field
34    let cargo_toml_path = project_path.join("Cargo.toml");
35    
36    if !cargo_toml_path.exists() {
37        result.add_error("Cargo.toml not found");
38        result.set_duration(start.elapsed());
39        return Ok(result);
40    }
41    
42    let contents = fs::read_to_string(&cargo_toml_path).await?;
43    let manifest: toml::Value = toml::from_str(&contents)
44        .map_err(|e| crate::Error::parse(format!("Failed to parse Cargo.toml: {}", e)))?;
45    
46    result.set_duration(start.elapsed());
47    
48    // Check for license field
49    let license = manifest
50        .get("package")
51        .and_then(|p| p.get("license"))
52        .and_then(|l| l.as_str());
53    
54    let license_file = manifest
55        .get("package")
56        .and_then(|p| p.get("license-file"))
57        .and_then(|l| l.as_str());
58    
59    if license.is_none() && license_file.is_none() {
60        result.add_error("No license specified in Cargo.toml");
61        result.add_suggestion("Add 'license = \"MIT OR Apache-2.0\"' to [package] section");
62        result.add_suggestion("Or add 'license-file = \"LICENSE\"' if using custom license");
63    } else if let Some(license_str) = license {
64        // Validate license string
65        let approved_licenses = [
66            "MIT",
67            "Apache-2.0", 
68            "MIT OR Apache-2.0",
69            "Apache-2.0 OR MIT",
70            "BSD-3-Clause",
71            "BSD-2-Clause",
72            "ISC",
73            "MPL-2.0",
74        ];
75        
76        if !approved_licenses.iter().any(|&approved| license_str.contains(approved)) {
77            result.add_error(format!("Uncommon license detected: {}", license_str));
78            result.add_suggestion("Consider using a standard license like 'MIT OR Apache-2.0'");
79            result.add_context("This may cause issues with some package managers");
80        } else {
81            result.add_context(format!("License: {}", license_str));
82        }
83        
84        // Check if license file exists for certain licenses
85        if license_str.contains("MIT") || license_str.contains("Apache") {
86            let license_files = ["LICENSE", "LICENSE.txt", "LICENSE.md", "LICENSE-MIT", "LICENSE-APACHE"];
87            let has_license_file = license_files.iter()
88                .any(|&file| project_path.join(file).exists());
89            
90            if !has_license_file {
91                result.add_error("License specified but no LICENSE file found");
92                result.add_suggestion("Create a LICENSE file with the license text");
93            }
94        }
95    }
96    
97    // Check for other important metadata
98    let description = manifest
99        .get("package")
100        .and_then(|p| p.get("description"))
101        .and_then(|d| d.as_str());
102    
103    if description.is_none() || description.unwrap().trim().is_empty() {
104        result.add_error("Missing or empty description in Cargo.toml");
105        result.add_suggestion("Add a clear description of what your crate does");
106    }
107    
108    let repository = manifest
109        .get("package")
110        .and_then(|p| p.get("repository"))
111        .and_then(|r| r.as_str());
112    
113    if repository.is_none() {
114        result.add_error("Missing repository URL in Cargo.toml");
115        result.add_suggestion("Add 'repository = \"https://github.com/user/repo\"' to [package]");
116    }
117    
118    Ok(result)
119}

121#[cfg(test)]
122#[allow(clippy::unwrap_used, clippy::expect_used)]
123mod tests {
124    use super::*;
125    use tempfile::TempDir;
126    
127    #[tokio::test]
128    async fn test_license_check_with_valid_license() {
129        let temp_dir = TempDir::new().unwrap();
130        
131        let cargo_toml = r#"
132[package]
133name = "test"
134version = "0.1.0"
135edition = "2021"
136license = "MIT OR Apache-2.0"
137description = "A test crate"
138repository = "https://github.com/user/test"
139"#;
140        
141        fs::write(temp_dir.path().join("Cargo.toml"), cargo_toml).await.unwrap();
142        
143        let result = run(temp_dir.path()).await.unwrap();
144        
145        // Should pass with valid license
146        assert!(result.passed);
147    }
148    
149    #[test]
150    fn test_license_check_struct() {
151        assert_eq!(LicenseCheck::name(), "license");
152        assert!(!LicenseCheck::description().is_empty());
153    }

11/// License check implementation
12pub struct LicenseCheck;
13
14impl SafetyCheck for LicenseCheck {
15    async fn run(project_path: &Path) -> Result<CheckResult> {
16        run(project_path).await
17    }
18
19    fn name() -> &'static str {
20        "license"
21    }
22
23    fn description() -> &'static str {
24        "Validates license compatibility and presence"
25    }
26}

28/// Validate license configuration
29pub async fn run(project_path: &Path) -> Result<CheckResult> {
30    let start = Instant::now();
31    let mut result = CheckResult::new(CheckType::License);
32
33    let manifest = match load_cargo_manifest(project_path).await {
34        Ok(manifest) => manifest,
35        Err(e) => {
36            result.add_error(&e.to_string());
37            result.set_duration(start.elapsed());
38            return Ok(result);
39        }
40    };
41
42    validate_license_fields(&manifest, project_path, &mut result);
43    validate_metadata_fields(&manifest, &mut result);
44
45    result.set_duration(start.elapsed());
46    Ok(result)
47}
48
49/// Load and parse Cargo.toml manifest
50async fn load_cargo_manifest(project_path: &Path) -> Result<toml::Value> {
51    let cargo_toml_path = project_path.join("Cargo.toml");
52
53    if !cargo_toml_path.exists() {
54        return Err(crate::Error::io("Cargo.toml not found".to_string()));
55    }
56
57    let contents = fs::read_to_string(&cargo_toml_path).await?;
58    let manifest: toml::Value = toml::from_str(&contents)
59        .map_err(|e| crate::Error::parse(format!("Failed to parse Cargo.toml: {}", e)))?;
60
61    Ok(manifest)
62}
63
64/// Validate license-related fields
65fn validate_license_fields(manifest: &toml::Value, project_path: &Path, result: &mut CheckResult) {
66    let license = manifest
67        .get("package")
68        .and_then(|p| p.get("license"))
69        .and_then(|l| l.as_str());
70
71    let license_file = manifest
72        .get("package")
73        .and_then(|p| p.get("license-file"))
74        .and_then(|l| l.as_str());
75
76    if license.is_none() && license_file.is_none() {
77        result.add_error("No license specified in Cargo.toml");
78        result.add_suggestion("Add 'license = \"MIT OR Apache-2.0\"' to [package] section");
79        result.add_suggestion("Or add 'license-file = \"LICENSE\"' if using custom license");
80    } else if let Some(license_str) = license {
81        validate_license_string(license_str, result);
82        check_license_file_exists(license_str, project_path, result);
83    }
84}
85
86/// Validate license string against approved licenses
87fn validate_license_string(license_str: &str, result: &mut CheckResult) {
88    let approved_licenses = [
89        "MIT",
90        "Apache-2.0",
91        "MIT OR Apache-2.0",
92        "Apache-2.0 OR MIT",
93        "BSD-3-Clause",
94        "BSD-2-Clause",
95        "ISC",
96        "MPL-2.0",
97    ];
98
99    if !approved_licenses
100        .iter()
101        .any(|&approved| license_str.contains(approved))
102    {
103        result.add_error(format!("Uncommon license detected: {}", license_str));
104        result.add_suggestion("Consider using a standard license like 'MIT OR Apache-2.0'");
105        result.add_context("This may cause issues with some package managers");
106    } else {
107        result.add_context(format!("License: {}", license_str));
108    }
109}
110
111/// Check if license file exists for licenses that require it
112fn check_license_file_exists(license_str: &str, project_path: &Path, result: &mut CheckResult) {
113    if license_str.contains("MIT") || license_str.contains("Apache") {
114        let license_files = [
115            "LICENSE",
116            "LICENSE.txt",
117            "LICENSE.md",
118            "LICENSE-MIT",
119            "LICENSE-APACHE",
120        ];
121
122        let has_license_file = license_files
123            .iter()
124            .any(|&file| project_path.join(file).exists());
125
126        if !has_license_file {
127            result.add_error("License specified but no LICENSE file found");
128            result.add_suggestion("Create a LICENSE file with the license text");
129        }
130    }
131}
132
133/// Validate other metadata fields
134fn validate_metadata_fields(manifest: &toml::Value, result: &mut CheckResult) {
135    let description = manifest
136        .get("package")
137        .and_then(|p| p.get("description"))
138        .and_then(|d| d.as_str());
139
140    if description.is_none_or(|d| d.trim().is_empty()) {
141        result.add_error("Missing or empty description in Cargo.toml");
142        result.add_suggestion("Add a clear description of what your crate does");
143    }
144
145    let repository = manifest
146        .get("package")
147        .and_then(|p| p.get("repository"))
148        .and_then(|r| r.as_str());
149
150    if repository.is_none() {
151        result.add_error("Missing repository URL in Cargo.toml");
152        result.add_suggestion("Add 'repository = \"https://github.com/user/repo\"' to [package]");
153    }
154}

162    async fn test_license_check_with_valid_license() {
163        let temp_dir = TempDir::new().unwrap();
164
165        let cargo_toml = r#"
166[package]
167name = "test"
168version = "0.1.0"
169edition = "2021"
170license = "MIT OR Apache-2.0"
171description = "A test crate"
172repository = "https://github.com/user/test"
173"#;
174
175        fs::write(temp_dir.path().join("Cargo.toml"), cargo_toml)
176            .await
177            .unwrap();
178
179        // Create LICENSE file
180        fs::write(
181            temp_dir.path().join("LICENSE"),
182            "MIT License\n\nCopyright (c) 2025",
183        )
184        .await
185        .unwrap();
186
187        let result = run(temp_dir.path()).await.unwrap();
188
189        // Debug: Print what errors were found
190        if !result.passed {
191            for error in &result.errors {
192                println!("License test error: {}", error);

194        }
195
196        // Should pass with valid license
197        assert!(result.passed);
198    }

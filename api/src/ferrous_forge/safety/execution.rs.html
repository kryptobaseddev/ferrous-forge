17/// Progress display and execution coordinator
18pub struct ExecutionManager {
19    /// Whether to show progress indicators
20    show_progress: bool,
21    /// Whether to run checks in parallel
22    parallel_checks: bool,
23}
24
25impl ExecutionManager {
26    /// Create a new execution manager
27    pub fn new(show_progress: bool, parallel_checks: bool) -> Self {
28        Self {
29            show_progress,
30            parallel_checks,
31        }
32    }

34    /// Setup progress indicators and display header
35    pub fn setup_progress_display(&self, stage: PipelineStage) -> Option<MultiProgress> {
36        let multi_progress = if self.show_progress {
37            Some(MultiProgress::new())
38        } else {
39            None
40        };
41
42        println!("üõ°Ô∏è  Ferrous Forge Safety Pipeline - {}", stage.display_name());
43        println!("{}", "=".repeat(50));
44
45        multi_progress
46    }
47
48    /// Execute checks for the stage
49    pub async fn execute_stage_checks(
50        &self,
51        stage_config: &StageConfig,
52        report: &mut SafetyReport,
53        multi_progress: Option<&MultiProgress>,
54        project_path: &Path,
55    ) -> Result<()> {
56        let checks = &stage_config.checks;
57
58        if self.parallel_checks && checks.len() > 1 {
59            self.run_checks_parallel(checks, report, multi_progress, project_path)
60                .await
61        } else {
62            self.run_checks_sequential(checks, report, multi_progress, project_path)
63                .await
64        }
65    }
66
67    /// Run checks sequentially with progress indicators
68    async fn run_checks_sequential(
69        &self,
70        checks: &[CheckType],
71        report: &mut SafetyReport,
72        multi_progress: Option<&MultiProgress>,
73        project_path: &Path,
74    ) -> Result<()> {
75        for check_type in checks {
76            let pb = if let Some(mp) = multi_progress {
77                let pb = mp.add(ProgressBar::new_spinner());
78                pb.set_style(
79                    ProgressStyle::default_spinner()
80                        .template("{spinner:.green} {msg}")
81                        .unwrap_or_else(|_| ProgressStyle::default_spinner()),
82                );
83                pb.set_message(format!("Running {}...", check_type.display_name()));
84                pb.enable_steady_tick(Duration::from_millis(100));
85                Some(pb)
86            } else {
87                None
88            };
89
90            let check_result = execute_check(*check_type, project_path).await?;
91
92            if let Some(pb) = pb {
93                pb.finish_with_message(format!(
94                    "{} {} ({:.2}s)",
95                    check_result.status_emoji(),
96                    check_type.display_name(),
97                    check_result.duration.as_secs_f64()
98                ));
99            } else {
100                println!(
101                    "  {} {} ({:.2}s)",
102                    check_result.status_emoji(),
103                    check_type.display_name(),
104                    check_result.duration.as_secs_f64()
105                );
106            }
107
108            report.add_check(check_result);
109        }
110
111        Ok(())
112    }
113
114    /// Run checks in parallel for better performance
115    async fn run_checks_parallel(
116        &self,
117        checks: &[CheckType],
118        report: &mut SafetyReport,
119        _multi_progress: Option<&MultiProgress>,
120        project_path: &Path,
121    ) -> Result<()> {
122        // For now, implement as sequential until we add proper parallel execution
123        // Parallel execution requires careful handling of stdout/stderr
124        self.run_checks_sequential(checks, report, _multi_progress, project_path)
125            .await
126    }
127}

129/// Execute a specific check type
130pub async fn execute_check(check_type: CheckType, project_path: &Path) -> Result<CheckResult> {
131    match check_type {
132        CheckType::Format => checks::format::run(project_path).await,
133        CheckType::Clippy => checks::clippy::run(project_path).await,
134        CheckType::Build => checks::build::run(project_path).await,
135        CheckType::Test => checks::test::run(project_path).await,
136        CheckType::Audit => checks::audit::run(project_path).await,
137        CheckType::Doc => checks::doc::run(project_path).await,
138        CheckType::PublishDryRun => checks::publish::run(project_path).await,
139        CheckType::Standards => checks::standards::run(project_path).await,
140        CheckType::DocCoverage => checks::doc::coverage_check(project_path).await,
141        CheckType::License => checks::license::run(project_path).await,
142        CheckType::Semver => checks::semver::run(project_path).await,
143    }
144}

146/// Handle check result and convert errors to check results
147pub fn handle_check_result(
148    check_result: Result<CheckResult>,
149    check_type: CheckType,
150) -> Result<CheckResult> {
151    match check_result {
152        Ok(result) => Ok(result),
153        Err(e) => {
154            let mut result = CheckResult::new(check_type);
155            result.add_error(format!("Check failed: {}", e));
156            result.add_suggestion("Check the error message above for details");
157            Ok(result)
158        }
159    }
160}

162/// Get the appropriate stage for a check type
163pub fn get_stage_for_check(check_type: CheckType) -> PipelineStage {
164    // Find the first stage that includes this check
165    for stage in [
166        PipelineStage::PreCommit,
167        PipelineStage::PrePush,
168        PipelineStage::Publish,
169    ] {
170        if CheckType::for_stage(stage).contains(&check_type) {
171            return stage;
172        }
173    }
174    PipelineStage::PreCommit // Default fallback
175}

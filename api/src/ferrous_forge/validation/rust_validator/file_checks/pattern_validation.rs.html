1//! Code pattern validation
2
3use super::super::patterns::{is_in_string_literal, ValidationPatterns};
4use crate::validation::{Severity, Violation, ViolationType};
5use crate::Result;
6use std::path::Path;
7
8/// Legacy large function that contains the main validation logic
9pub fn validate_patterns(
10    rust_file: &Path,
11    lines: &[&str],
12    patterns: &ValidationPatterns,
13    violations: &mut Vec<Violation>,
14    is_test_file: bool,
15    allow_unwrap: bool,
16    allow_expect: bool,
17) -> Result<()> {
18    let mut in_test_block = false;
19    let mut current_function_start: Option<usize> = None;
20    let mut next_function_is_test = false;
21    let mut in_test_module = false;
22    let mut next_module_is_test = false;
23    let mut brace_depth = 0;
24    let mut test_module_depth = None;
25
26    for (i, line) in lines.iter().enumerate() {
27        let line_stripped = line.trim();
28
29        // Track brace depth to properly scope test modules
30        brace_depth += line.chars().filter(|&c| c == '{').count();
31        brace_depth = brace_depth.saturating_sub(line.chars().filter(|&c| c == '}').count());
32
33        // Reset test module when we exit its scope
34        if let Some(depth) = test_module_depth {
35            if brace_depth <= depth {
36                in_test_module = false;
37                test_module_depth = None;
38            }
39        }
40
41        // Check for test module markers
42        if line_stripped.contains("#[cfg(test)]") {
43            next_module_is_test = true;
44        }
45
46        // Check if we're entering a test module
47        if (line_stripped.starts_with("mod tests")
48            || (line_stripped.starts_with("mod ") && line_stripped.contains("test")))
49            && next_module_is_test
50        {
51            in_test_module = true;
52            test_module_depth = Some(brace_depth);
53            next_module_is_test = false;
54        }
55
56        // Also handle case where #[cfg(test)] and mod are on same line
57        if line_stripped.contains("#[cfg(test)]") && line_stripped.contains("mod") {
58            in_test_module = true;
59            test_module_depth = Some(brace_depth);
60        }
61
62        // Check for test attributes that apply to the next function
63        if line_stripped.contains("#[test]")
64            || line_stripped.contains("#[tokio::test]")
65            || line_stripped.contains("#[bench]")
66        {
67            next_function_is_test = true;
68        }
69
70        // Track function boundaries and check function size
71        if patterns.function_def.is_match(line) {
72            // Check previous function size
73            if let Some(start) = current_function_start {
74                let func_lines = i - start;
75                if func_lines > 50 {
76                    violations.push(Violation {
77                        violation_type: ViolationType::FunctionTooLarge,
78                        file: rust_file.to_path_buf(),
79                        line: start + 1,
80                        message: format!(
81                            "Function has {} lines, maximum allowed is 50",
82                            func_lines
83                        ),
84                        severity: Severity::Error,
85                    });
86                }
87            }
88
89            // Set test block state based on whether this function is a test
90            in_test_block = next_function_is_test;
91            next_function_is_test = false; // Reset for next function
92            current_function_start = Some(i);
93        }
94
95        // Check for underscore bandaid coding
96        if patterns.underscore_param.is_match(line) {
97            let has_underscore = line.contains("_unused") || line.contains("_param");
98
99            // Enhanced string detection
100            let not_in_unused_literal = !is_in_string_literal(line, "_unused")
101                && !line.contains("r\"")
102                && !line.contains("r#\"");
103            let not_in_param_literal = !is_in_string_literal(line, "_param");
104
105            // Check if it's actually test content (ignore false positives)
106            let is_test_content = line.contains("test_function")
107                || line.contains("test_module")
108                || line.contains("assert!");
109
110            if has_underscore && not_in_unused_literal && not_in_param_literal && !is_test_content {
111                violations.push(Violation {
112                    violation_type: ViolationType::UnderscoreBandaid,
113                    file: rust_file.to_path_buf(),
114                    line: i + 1,
115                    message: "BANNED: Underscore parameter (_param) - \
116                             fix the design instead of hiding warnings"
117                        .to_string(),
118                    severity: Severity::Error,
119                });
120            }
121        }
122
123        // Check for unwrap in production code (not in tests or if allowed)
124        if !in_test_block
125            && !in_test_module
126            && !is_test_file
127            && !allow_unwrap
128            && !line_stripped.starts_with("//")
129            && patterns.unwrap_call.is_match(line)
130            && !is_in_string_literal(line, ".unwrap()")
131        {
132            violations.push(Violation {
133                violation_type: ViolationType::UnwrapInProduction,
134                file: rust_file.to_path_buf(),
135                line: i + 1,
136                message: "BANNED: .unwrap() in production code - use proper error handling with ?"
137                    .to_string(),
138                severity: Severity::Error,
139            });
140        }
141
142        // Check for expect in production code (not in tests or if allowed)
143        if !in_test_block
144            && !in_test_module
145            && !is_test_file
146            && !allow_expect
147            && !line_stripped.starts_with("//")
148            && patterns.expect_call.is_match(line)
149            && !is_in_string_literal(line, ".expect(")
150        {
151            violations.push(Violation {
152                violation_type: ViolationType::UnwrapInProduction,
153                file: rust_file.to_path_buf(),
154                line: i + 1,
155                message: "BANNED: .expect() in production code - use proper error handling with ?"
156                    .to_string(),
157                severity: Severity::Error,
158            });
159        }
160
161        // Check line lengths (100 character limit)
162        if line.len() > 100 {
163            violations.push(Violation {
164                violation_type: ViolationType::LineTooLong,
165                file: rust_file.to_path_buf(),
166                line: i + 1,
167                message: format!("Line has {} characters, maximum allowed is 100", line.len()),
168                severity: Severity::Warning,
169            });
170        }
171    }
172
173    Ok(())
174}

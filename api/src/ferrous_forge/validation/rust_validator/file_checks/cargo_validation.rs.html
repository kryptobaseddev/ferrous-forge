9/// Validates a `Cargo.toml` file for standards compliance (reads file itself).
10/// Used by tests and legacy callers.
11///
12/// # Errors
13///
14/// Returns an error if the file cannot be read.
15pub async fn validate_cargo_toml(
16    cargo_file: &Path,
17    violations: &mut Vec<Violation>,
18    required_edition: &str,
19    required_rust_version: &str,
20) -> Result<()> {
21    let content = fs::read_to_string(cargo_file).await?;
22    let lines: Vec<&str> = content.lines().collect();
23    validate_cargo_toml_content(
24        cargo_file,
25        &lines,
26        violations,
27        required_edition,
28        required_rust_version,
29    );
30    Ok(())
31}
32
33/// Validates `Cargo.toml` content that has already been read into lines.
34/// Used by `file_checks.rs` to avoid double file reads.
35pub fn validate_cargo_toml_content(
36    cargo_file: &Path,
37    lines: &[&str],
38    violations: &mut Vec<Violation>,
39    required_edition: &str,
40    required_rust_version: &str,
41) {
42    // Parse the manifest with the TOML parser so we correctly understand
43    // workspace inheritance (`version.workspace = true`, `edition = { workspace = true }`)
44    // and virtual workspaces. If parsing fails cargo itself will surface the error;
45    // don't double-report.
46    let content = lines.join("\n");
47    let parsed = match toml::from_str::<Value>(&content) {
48        Ok(v) => v,
49        Err(_) => return,
50    };
51
52    validate_edition(&parsed, lines, cargo_file, violations, required_edition);
53    validate_rust_version_field(
54        &parsed,
55        lines,
56        cargo_file,
57        violations,
58        required_rust_version,
59    );
60}

62/// Resolution of a package field after accounting for workspace inheritance
63enum ResolvedField<'a> {
64    /// Literal string value (e.g. `edition = "2024"`)
65    Value(&'a str),
66    /// `{ workspace = true }` — the member inherits from the workspace root
67    WorkspaceInherited,
68    /// Present but not a string and not workspace inheritance — malformed; skip
69    Other,
70    /// Not present at all
71    Absent,
72}
73
74fn classify_field(value: &Value) -> ResolvedField<'_> {
75    if let Some(s) = value.as_str() {
76        return ResolvedField::Value(s);
77    }
78    if let Some(tbl) = value.as_table()
79        && tbl.get("workspace").and_then(Value::as_bool) == Some(true)
80    {
81        return ResolvedField::WorkspaceInherited;
82    }
83    ResolvedField::Other
84}
85
86/// Resolve a field by checking `[package].<field>` then `[workspace.package].<field>`.
87fn resolve_field<'a>(parsed: &'a Value, field: &str) -> ResolvedField<'a> {
88    if let Some(value) = parsed.get("package").and_then(|p| p.get(field)) {
89        return classify_field(value);
90    }
91    if let Some(value) = parsed
92        .get("workspace")
93        .and_then(|w| w.get("package"))
94        .and_then(|p| p.get(field))
95    {
96        return classify_field(value);
97    }
98    ResolvedField::Absent
99}

101/// A pure virtual manifest has `[workspace]` but no `[package]` section.
102fn is_virtual_workspace(parsed: &Value) -> bool {
103    parsed.get("package").is_none() && parsed.get("workspace").is_some()
104}
105
106/// Locate the 1-indexed line of a field in the original source for error reporting.
107/// Matches both `edition = "..."` and `edition.workspace = true` forms.
108fn find_field_line(lines: &[&str], field: &str) -> usize {
109    let eq_key = format!("{field} =");
110    let tight_key = format!("{field}=");
111    let dotted_key = format!("{field}.workspace");
112    for (i, line) in lines.iter().enumerate() {
113        let trimmed = line.trim_start();
114        if trimmed.starts_with(&eq_key)
115            || trimmed.starts_with(&tight_key)
116            || trimmed.starts_with(&dotted_key)
117        {
118            return i + 1;
119        }
120    }
121    0
122}

124/// Check edition field against locked required value
125fn validate_edition(
126    parsed: &Value,
127    lines: &[&str],
128    cargo_file: &Path,
129    violations: &mut Vec<Violation>,
130    required_edition: &str,
131) {
132    match resolve_field(parsed, "edition") {
133        ResolvedField::Value(found_edition) => {
134            if found_edition != required_edition {
135                violations.push(Violation {
136                    violation_type: ViolationType::WrongEdition,
137                    file: cargo_file.to_path_buf(),
138                    line: find_field_line(lines, "edition"),
139                    message: format!(
140                        "FERROUS FORGE [LOCKED SETTING] — Edition Violation\n\
141                         ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━\n\
142                         \n\
143                           Found:    edition = \"{found_edition}\"\n\
144                           Required: edition = \"{required_edition}\"\n\
145                           Lock:     .ferrous-forge/config.toml → required_edition = \"{required_edition}\"\n\
146                         \n\
147                           ⚠ AI AGENT NOTICE: This edition is locked by project configuration.\n\
148                           DO NOT change required_edition without human approval.\n\
149                           DO NOT downgrade Cargo.toml to match a lower edition.\n\
150                         \n\
151                           To fix: Update edition = \"{required_edition}\" in Cargo.toml, then run:\n\
152                             cargo fix --edition"
153                    ),
154                    severity: Severity::Error,
155                });
156            }
157        }
158        ResolvedField::WorkspaceInherited => {
159            // Member inherits from workspace root; that root's manifest is
160            // validated separately when walking the project.
161        }
162        ResolvedField::Other => {
163            // Malformed TOML value — cargo itself will reject; don't double-error.
164        }
165        ResolvedField::Absent => {
166            // Virtual workspaces without an explicit [workspace.package].edition
167            // are valid in cargo; members then must set their own. Only report if
168            // this manifest actually defines a [package] section.
169            if !is_virtual_workspace(parsed) {
170                violations.push(Violation {
171                    violation_type: ViolationType::WrongEdition,
172                    file: cargo_file.to_path_buf(),
173                    line: 0,
174                    message: format!(
175                        "FERROUS FORGE [LOCKED SETTING] — Missing Edition\n\
176                         ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━\n\
177                         \n\
178                           Missing edition specification in Cargo.toml.\n\
179                           Required: edition = \"{required_edition}\"\n\
180                         \n\
181                           ⚠ AI AGENT NOTICE: Add edition = \"{required_edition}\" to [package] section."
182                    ),
183                    severity: Severity::Error,
184                });
185            }
186        }
187    }
188}

190/// Check rust-version field against locked required value
191fn validate_rust_version_field(
192    parsed: &Value,
193    lines: &[&str],
194    cargo_file: &Path,
195    violations: &mut Vec<Violation>,
196    required_rust_version: &str,
197) {
198    if required_rust_version.is_empty() {
199        return;
200    }
201
202    match resolve_field(parsed, "rust-version") {
203        ResolvedField::Value(found_version) => {
204            if found_version != required_rust_version {
205                violations.push(Violation {
206                    violation_type: ViolationType::OldRustVersion,
207                    file: cargo_file.to_path_buf(),
208                    line: find_field_line(lines, "rust-version"),
209                    message: format!(
210                        "FERROUS FORGE [LOCKED SETTING] — Rust Version Violation\n\
211                         ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━\n\
212                         \n\
213                           Found:    rust-version = \"{found_version}\"\n\
214                           Required: rust-version = \"{required_rust_version}\"\n\
215                           Lock:     .ferrous-forge/config.toml → required_rust_version = \"{required_rust_version}\"\n\
216                         \n\
217                           ⚠ AI AGENT NOTICE: This rust-version is locked by project configuration.\n\
218                           DO NOT change required_rust_version without human approval.\n\
219                           DO NOT downgrade rust-version to resolve compilation errors."
220                    ),
221                    severity: Severity::Error,
222                });
223            }
224        }
225        // Workspace-inherited or absent → nothing to validate here.
226        // `rust-version` is optional in cargo so missing is acceptable.
227        ResolvedField::WorkspaceInherited | ResolvedField::Other | ResolvedField::Absent => {}
228    }

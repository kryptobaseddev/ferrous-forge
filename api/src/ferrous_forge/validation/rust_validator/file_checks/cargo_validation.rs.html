8/// Validates a `Cargo.toml` file for standards compliance (reads file itself).
9/// Used by tests and legacy callers.
10///
11/// # Errors
12///
13/// Returns an error if the file cannot be read.
14pub async fn validate_cargo_toml(
15    cargo_file: &Path,
16    violations: &mut Vec<Violation>,
17    required_edition: &str,
18    required_rust_version: &str,
19) -> Result<()> {
20    let content = fs::read_to_string(cargo_file).await?;
21    let lines: Vec<&str> = content.lines().collect();
22    validate_cargo_toml_content(
23        cargo_file,
24        &lines,
25        violations,
26        required_edition,
27        required_rust_version,
28    );
29    Ok(())
30}
31
32/// Validates `Cargo.toml` content that has already been read into lines.
33/// Used by `file_checks.rs` to avoid double file reads.
34pub fn validate_cargo_toml_content(
35    cargo_file: &Path,
36    lines: &[&str],
37    violations: &mut Vec<Violation>,
38    required_edition: &str,
39    required_rust_version: &str,
40) {
41    validate_edition(lines, cargo_file, violations, required_edition);
42    validate_rust_version_field(lines, cargo_file, violations, required_rust_version);
43}

45/// Check edition field against locked required value
46fn validate_edition(
47    lines: &[&str],
48    cargo_file: &Path,
49    violations: &mut Vec<Violation>,
50    required_edition: &str,
51) {
52    let mut edition_found = false;
53
54    for (i, line) in lines.iter().enumerate() {
55        if line.contains("edition") && line.contains('=') {
56            edition_found = true;
57            let found_edition = extract_quoted_value(line);
58
59            if found_edition.as_deref() != Some(required_edition) {
60                let found_display = found_edition.as_deref().unwrap_or("<unknown>");
61                violations.push(Violation {
62                    violation_type: ViolationType::WrongEdition,
63                    file: cargo_file.to_path_buf(),
64                    line: i + 1,
65                    message: format!(
66                        "FERROUS FORGE [LOCKED SETTING] — Edition Violation\n\
67                         ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━\n\
68                         \n\
69                           Found:    edition = \"{found_display}\"\n\
70                           Required: edition = \"{required_edition}\"\n\
71                           Lock:     .ferrous-forge/config.toml → required_edition = \"{required_edition}\"\n\
72                         \n\
73                           ⚠ AI AGENT NOTICE: This edition is locked by project configuration.\n\
74                           DO NOT change required_edition without human approval.\n\
75                           DO NOT downgrade Cargo.toml to match a lower edition.\n\
76                         \n\
77                           To fix: Update edition = \"{required_edition}\" in Cargo.toml, then run:\n\
78                             cargo fix --edition"
79                    ),
80                    severity: Severity::Error,
81                });
82            }
83            break;
84        }
85    }
86
87    if !edition_found {
88        violations.push(Violation {
89            violation_type: ViolationType::WrongEdition,
90            file: cargo_file.to_path_buf(),
91            line: 0,
92            message: format!(
93                "FERROUS FORGE [LOCKED SETTING] — Missing Edition\n\
94                 ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━\n\
95                 \n\
96                   Missing edition specification in Cargo.toml.\n\
97                   Required: edition = \"{required_edition}\"\n\
98                 \n\
99                   ⚠ AI AGENT NOTICE: Add edition = \"{required_edition}\" to [package] section."
100            ),
101            severity: Severity::Error,
102        });
103    }
104}

106/// Check rust-version field against locked required value
107fn validate_rust_version_field(
108    lines: &[&str],
109    cargo_file: &Path,
110    violations: &mut Vec<Violation>,
111    required_rust_version: &str,
112) {
113    if required_rust_version.is_empty() {
114        return;
115    }
116
117    for (i, line) in lines.iter().enumerate() {
118        if line.contains("rust-version") && line.contains('=') {
119            let found_version = extract_quoted_value(line);
120
121            if found_version.as_deref() != Some(required_rust_version) {
122                let found_display = found_version.as_deref().unwrap_or("<unknown>");
123                violations.push(Violation {
124                    violation_type: ViolationType::OldRustVersion,
125                    file: cargo_file.to_path_buf(),
126                    line: i + 1,
127                    message: format!(
128                        "FERROUS FORGE [LOCKED SETTING] — Rust Version Violation\n\
129                         ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━\n\
130                         \n\
131                           Found:    rust-version = \"{found_display}\"\n\
132                           Required: rust-version = \"{required_rust_version}\"\n\
133                           Lock:     .ferrous-forge/config.toml → required_rust_version = \"{required_rust_version}\"\n\
134                         \n\
135                           ⚠ AI AGENT NOTICE: This rust-version is locked by project configuration.\n\
136                           DO NOT change required_rust_version without human approval.\n\
137                           DO NOT downgrade rust-version to resolve compilation errors."
138                    ),
139                    severity: Severity::Error,
140                });
141            }
142            return;
143        }
144    }
145    // rust-version absent from Cargo.toml is acceptable
146}

148/// Extract a quoted string value from a TOML line like: key = "value"
149fn extract_quoted_value(line: &str) -> Option<String> {
150    let after_eq = line.split('=').nth(1)?.trim();
151    let inner = after_eq.trim_matches('"').trim_matches('\'').trim();
152    if inner.is_empty() {
153        None
154    } else {
155        Some(inner.to_string())
156    }
157}
